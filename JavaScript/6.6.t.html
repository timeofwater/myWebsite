    div, li, ul {

    .box {
        display: block;
        margin: 50px;
    }

    .star {
        display: block;
        width: 9px;
        height: 10px;
        background: url("../pictures/star.png") no-repeat;
        cursor: pointer;
    }

    .on {
        background-position: 0 -10px;

    <body>
    <div class="box">
        <ul id="ul">
            <li><span class="star"></span></li>
            <li><span class="star"></span></li>
            <li><span class="star"></span></li>
            <li><span class="star"></span></li>
            <li><span class="star"></span></li>
        </ul>
    </div>

    <script type="text/javascript">
        let aStar = document.getElementsByClassName("star");
        let objectUl = document.getElementById("ul");
        let length = aStar.length;
        let score = -1;

        for (let i = 0; i < length; i++) {
            aStar[i].onmouseenter = function () {
                for (let j = 0; j < length; j++) {
                    aStar[j].className = (j <= i) ? "star on" : "star";
                }
            }
            aStar[i].onclick = function () {
                score = i;
            }
            objectUl.onmouseleave = function () {
                for (let j = 0; j < length; j++) {
                    aStar[j].className = (j <= score) ? "star on" : "star";
                }
            }
        }
    </script>
    </body>
