CSS JS 周期表 目标收集 
with:

    with(对象){
        对象操作;
    }
    在with语句里, object可省略

    
        let x = Math.cos(2 * Math.PI) + Math.sin(Math.LN10);

        with (Math) {
            let x = cos(2 * PI) + sin(LN10);
        }
    

for-in:

    for(变量 in 对象){
        语句段;
    }
    对某对象的所有属性进行循环操作, 而不需得知属性个数

    
        let student = {
            name: "Tom",
            gender: "male",
            age: 18
        };

        for(let key in student) {
            document.write("student." + key + " = " + student[key] + "<br>");
        }

this:
    对当前对象的引用

    this = {document        ,主程序中
           |object          ,with语句块中
           |调用函数的对象  ,函数中
           {发生事件的对象  ,事件处理程序中

new:
    创建指定对象的一个实例

    let day = new Date();

in:
    用于判断对象是否存在某个属性

    "属性名" in 对象名;      //返回一个boolean

delete:

    delete 对象.属性名;

    
        let person = {
            name: "Tom",
            age: 18
        };
        for (let key in person) {
            document.write(key + "<br>    ");
        }

        delete person.age;
        document.write("<br>");
        for (let key in person) {
            document.write("        " + key + "<br>");
        }
