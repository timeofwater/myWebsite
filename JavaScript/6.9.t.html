    //
    let obBody = document.getElementById("b");
    obBody.addEventListener("click", click, true);
    let number = 0;

    function click(event) {
        let obSpan = document.createElement("span");
        obSpan.setAttribute("class", "float");
        obSpan.setAttribute("id", "" + number);
        obBody.appendChild(obSpan);

        //内容
        let arr = ["Java", "C", "C++", "PHP", "Python", "Android", "JavaScript", "JSON", "XML", "MySQL"];
        document.getElementById("" + number).innerHTML = arr[number];
        number = (number + 1) % arr.length;

        //动态效果
        let x = event.pageX;
        let y = event.pageY;
        let op = 100;
        let r = 100 + 155 * Math.random();
        let g = 100 + 155 * Math.random();
        let b = 100 + 155 * Math.random();

        let move = setInterval(function () {
            obSpan.setAttribute("style", "opacity:" + op + "%;" + "top:" + y + "px;" + "left:" + x + "px;" + "color: rgb(" + ~~r + "," + ~~g + "," + ~~b + ");");
            y -= 2;
            op -= 0.5;
        }, 5);
        setTimeout(function () {
            clearInterval(move);
            obBody.removeChild(obSpan);
        }, 1000);
    }
